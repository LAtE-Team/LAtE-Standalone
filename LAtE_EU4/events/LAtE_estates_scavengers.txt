		}

	mean_time_to_happen = {
		months = 600

			factor = 1.1

		}

		modifier = {
			factor = 0.9
			owner = { estate_loyalty = { estate = estates_scavengers loyalty = 50 } }
		}

		modifier = {
			factor = 0.9
			owner = { estate_loyalty = { estate = estates_scavengers loyalty = 75 } }
		}

		modifier = {
			factor = 0.9
			is_prosperous = yes
		}

		modifier = {
			factor = 0.9
			development = 30
		}

		modifier = {
			factor = 0.8
			owner = { has_estate_privilege = estate_scavengers_raise_payroll }
		}

	option = {
		name = estates_scavengers.3.a

	option = {
		name = estates_scavengers.3.b
		add_treasury = 100
		hidden_effect = { add_province_modifier = { name = estate_scavengers_province_empty duration = 3650 hidden = yes } }
	}
}
