# Centralize or Decentralize
country_event = {
	id = ordinate_flavor.1
	title = ordinate_flavor.1.t
	desc = ordinate_flavor.1.d
	picture = RELIGION_eventPicture

	is_triggered_only = yes

	trigger = {
		NOT = { has_country_modifier = ordinate_debate_modifier_1a }
		NOT = { has_country_modifier = ordinate_debate_modifier_1b }
	}

	option = {
		name = ordinate_flavor.1.a #Centralize the orders
		add_country_modifier = {
			name = ordinate_debate_modifier_1a
			duration = -1
		}
	}

	option = {
		name = ordinate_flavor.1.b #Leave them decentralized
		add_country_modifier = {
			name = ordinate_debate_modifier_1b
			duration = -1
		}
	}
}

# All debates done, get a stab or temporary missionary strength
country_event = {
	id = ordinate_flavor.10
	title = ordinate_flavor.10.t
	desc = ordinate_flavor.10.d
	picture = RELIGION_eventPicture

	is_triggered_only = yes

	option = {
		name = ordinate_flavor.10.a
		trigger = {
			NOT = { stability = 3 }
		}
		add_stability = 1
	}

	option = {
		name = ordinate_flavor.10.b
		trigger = {
			NOT = { has_country_modifier = ordinate_debate_modifier_10 }
		}
		add_country_modifier = {
			name = ordinate_debate_modifier_10
			duration = 3650
		}
	}

	option = {
		name = ordinate_flavor.10.c #You shouldn't ever see this
		trigger = {
			AND = {
				stability = 3
				has_country_modifier = ordinate_debate_modifier_10
			}
		}
	}
}
