# The Cowboys break free!
country_event = {
	id = LAtE_estate_disasters.1
	title = LAtE_estate_disasters.1.t
	desc = LAtE_estate_disasters.1.d
	picture = COSSACK_ESTATE_UPSET_eventPicture
	
	major = yes
	is_triggered_only = yes
	
	immediate = {
		set_country_flag = cowboy_estate_triggered
	}
	
	option = {
		name = LAtE_estate_disasters.1.a	
		add_country_modifier = {
			name = "serf_desertion"
			duration = 5475
		}
		create_independent_estate = {
			estate = estate_cowboys
			government = republic
			government_reform = cossacks_reform
			custom_national_ideas = cossack_ideas
			play_as = no
		}
	}
	
	option = {
		name = LAtE_estate_disasters.1.b
		trigger = { 
			ai = no 
		}
		add_country_modifier = {
			name = "serf_desertion"
			duration = 5475
		}
		create_independent_estate = {
			estate = estate_cowboys
			government = republic
			government_reform = cossacks_reform
			custom_national_ideas = cossack_ideas
			play_as = yes
		}
	}
}
