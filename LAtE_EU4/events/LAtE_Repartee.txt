			OR = {
				religion_group = afro_syncretic

			}
		}
		desc = repartee.1.latin.d
	}
	picture = RELIGION_eventPicture

	trigger = {
		LAtE_repartee_contender_trigger = yes

			has_discovered = ROOT
		}
	}

	fire_only_once = yes

	mean_time_to_happen = {
		months = 4
	}

	immediate = {
		set_global_flag = repartee_begins_flag
		REB = {
			set_variable = {
				which = catholicReparteeVariable
				value = 0
			}
			set_variable = {
				which = canadianReparteeVariable
				value = 0
			}
			set_variable = {
				which = protestantReparteeVariable
				value = 0
			}
			set_variable = {
				which = oldworldReparteeVariable
				value = 0
			}
			set_variable = {
				which = latinReparteeVariable
				value = 0
			}
		}
	}

	option = {
		name = repartee.1.catholic.o
		trigger = {
			religion = catholic
		}
		REB = {
			change_variable = {
				which = catholicReparteeVariable
				value = 1
			}
		}
	}

	option = {
		name = repartee.1.canadian.o
		trigger = {
			OR = {
				religion = ursuline
				religion = anglican
			}	
		}
		REB = {
			change_variable = {
				which = canadianReparteeVariable
				value = 1
			}
		}
	}

	option = {
		name = repartee.1.protestant.o
		trigger = {
			religion = protestant
		}
		REB = {
			change_variable = {
				which = protestantReparteeVariable
				value = 1
			}
		}
	}

	option = {
		name = repartee.1.oldworld.o
		trigger = {
			religion_group = oldworld
		}
		REB = {
			change_variable = {
				which = oldworldReparteeVariable
				value = 1
			}
		}
	}

	option = {
		name = repartee.1.latin.o
		trigger = {
			OR = {
				religion_group = afro_syncretic
				religion_group = cult_of_saints
			}
		}
		REB = {
			change_variable = {
				which = latinReparteeVariable
				value = 1
			}
		}
	}

	after = {
		every_country = {
			limit = {
				LAtE_repartee_contender_trigger = yes
			}
			country_event = {
				id = repartee.2
				days = 60 
			}
		}
	}
}
