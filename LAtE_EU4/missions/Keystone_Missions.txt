KEY_missions_accept_anabaptists = {
	slot = 5                    # Which column the missions will appear in. 1 to 5.
    generic = no             # Whether missions within this series are considered generic.
    ai = yes                  # Whether the AI will claim missions in this series.
    has_country_shield = yes  # Whether to display the country shield on the icon.    
    
    # Determines when a series appears for a country. Country scope.
    potential = {
        NOT = { map_setup = map_setup_random }
		OR = {
			AND = {
				primary_culture = deitsch
				capital_scope = {
					region = atlantica_region
				}
			}
			tag = DEI
			tag = AGH
			tag = KEY
		}
		has_country_flag = anabaptists_accepted
    }
    
    KEY_mission_allow_growth = {
        icon = mission_unite_home_region
        position = 1
                
        trigger = {
            has_estate_privilege = estate_anabaptists_allow_growth
        }
        
        effect = {
			add_country_modifier = {
				name = KEY_anabaptists_accept_modifier
				duration = 7300
			}
			add_estate_loyalty_modifier = {
				estate = estate_anabaptists
				desc = EST_VAL_ANB_KEY
				loyalty = 10
				duration = 7300
			}
        }
    }

    KEY_mission_anabaptists_prosper = {
        icon = mission_unite_home_region
		required_missions = {
			KEY_mission_allow_growth
		}

		provinces_to_highlight = {
			owned_by = ROOT
			NOT = {
				OR = {
					has_province_modifier = anabaptists_present_large
					has_province_modifier = anabaptists_present_medium
				}
			}
		}
                
        trigger = {
			calc_true_if = {
				amount = 10
				all_owned_province = {
					OR = {
						has_province_modifier = anabaptists_present_large
						has_province_modifier = anabaptists_present_medium
					}
				}
			}
        }
        
        effect = {
			add_years_of_income = 0.5
			every_owned_province = {
				limit = {
					OR = {
						has_province_modifier = anabaptists_present_large
						has_province_modifier = anabaptists_present_medium
					}
				}
				add_province_modifier = {
					name = KEY_anabaptists_growth_modifier
					duration = 7300
				}
			}
        }
    }
}
