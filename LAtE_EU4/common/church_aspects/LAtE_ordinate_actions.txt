		factor = 1
	}

ordinate_empower_ursulines = {
	sprite = "GFX_rust_cult_baubles_icon"
	cost = 100
	potential = {
		NOT = { has_country_flag = anabaptist_holy_order_flag }
	}
	trigger = {
		any_owned_province = {
			holy_order = ursuline_order
		}
		OR = {
			adm_power = 50
			has_estate = estate_church
		}
		NOT = {
			any_owned_province = {
				has_province_modifier = ordinate_empowered_ursulines_modifier
			}
		}
	}
	effect = {
		if = {
			limit = {
				has_estate = estate_church
			}
			add_estate_influence_modifier = {
				estate = estate_church
				desc = EST_EMPOWERED_URS_TT
				influence = 10
				duration = 3650
			}
		}
		else = {
			add_adm_power = -50
		}
		hidden_effect = {
			every_owned_province = {
				limit = {
					holy_order = ursuline_order
				}
				add_province_modifier = {
					name = ordinate_empowered_ursulines_modifier
					duration = 3650
				}
			}
		}
		custom_tooltip = ordinate_empower_ursulines_tt
	}
	ai_will_do = {
		factor = 1
		modifier = {
			factor = 0
			NOT = { 
				estate_influence = {
					estate = estate_church
					influence = 80
				}
			}
		}
	}
}
