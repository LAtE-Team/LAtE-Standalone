estate_press_disaster = {
	potential = {
		has_estate = estate_press
		estate_influence = {
			estate = estate_press
			influence = 80
		}
	}


	can_start = {
		has_any_disaster = no
		estate_influence = {
			estate = estate_press
			influence = 100
		}
		is_at_war = no
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = {
				estate_influence = {
					estate = estate_press
					influence = 100
				}		
			}
		}
	}
	
	down_progress = {
		factor = 0
		modifier = {
			factor = 1
			NOT = {
				estate_influence = {
					estate = estate_press
					influence = 100
				}
			}
		}
	}
	
	progress = {
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_press
					loyalty = 50
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_press
					loyalty = 40
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_press
					loyalty = 30
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_press
					loyalty = 20
				}
			}
			is_at_war = no
		}	
	}
	
	can_end = {
		custom_trigger_tooltip = {
			tooltip = EST_CRUSHED_PRESS
			OR = {
				NOT = { has_country_flag = press_estate_in_power }
				has_country_flag = army_estate_in_power
				has_country_flag = antiquarians_estate_in_power
			}						
		}
	}
	
	modifier = {	
		global_regiment_cost = 0.2
		legitimacy = -1
		global_missionary_strength = -1
	}
	
	stability_hit_on_war = 1

	on_start = LAtE_estate_disasters.24.s
	on_end = LAtE_estate_disasters.24.e
	
	on_monthly = {
	}
}
