estate_cowboys_disaster = {
	potential = {
		has_estate = estate_cowboys
		estate_influence = {
			estate = estate_cowboys
			influence = 80
		}
		OR = {
			NOT = { has_country_flag = cowboy_estate_triggered }
			had_country_flag = {
				flag = cowboy_estate_triggered
				days = 365
			}
		}
	}


	can_start = {
		has_any_disaster = no
		estate_influence = {
			estate = estate_cowboys
			influence = 100
		}
		is_at_war = no
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = {
				estate_influence = {
					estate = estate_cowboys
					influence = 100
				}		
			}
		}
	}
	
	down_progress = {
		factor = 0
		modifier = {
			factor = 1
			NOT = {
				estate_influence = {
					estate = estate_cowboys
					influence = 100
				}
			}
		}
	}
	
	progress = {
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_cowboys
					loyalty = 50
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_cowboys
					loyalty = 40
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_cowboys
					loyalty = 30
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_cowboys
					loyalty = 20
				}
			}
			is_at_war = no
		}
	}
	
	can_end = {
		always = yes
	}

	on_start = LAtE_estate_disasters.12.s
	
	on_start_effect = {
		if = {
			limit = {
				has_dlc = "The Cossacks"
			}
			custom_tooltip = estate_cowboys_disaster_tooltip_cowboys
		}
		else = {
			custom_tooltip = estate_cowboys_disaster_tooltip
		}
	}
	
	on_monthly = {
	}
}
